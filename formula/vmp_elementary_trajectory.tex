        h(x_0) = y_0 - f_0,\ \dot{h}(x_0) = \dot{y}_0 - \dot{f}_0,\ \ddot{h}(x_0) = \ddot{y}_0 - \ddot{f}_0 \\
        h(x_1) = y_1 - f_1,\ \dot{h}(x_1) = \dot{y}_1 - \dot{f}_1,\ \ddot{h}(x_1) = \ddot{y}_1 - \ddot{f}_1
        \end{matrix}
\end{equation}
